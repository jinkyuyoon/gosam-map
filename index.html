let boatMarker = null; function startGPS() { if (!navigator.geolocation) { document.getElementById('info').innerHTML = '❌ 이 브라우저는 GPS를 지원하지 않습니다.'; return; } document.getElementById('info').innerHTML = ` 📍 GPS 위치를 가져오는 중...
위치 권한을 허용해주세요 `; navigator.geolocation.watchPosition( function(position) { const lat = position.coords.latitude; const lng = position.coords.longitude; document.getElementById('info').innerHTML = ` 📍 현재 위치
위도: ${lat.toFixed(6)}
경도: ${lng.toFixed(6)}
정확도: ${Math.round(position.coords.accuracy)}m `; if (!boatMarker) { boatMarker = L.marker([lat, lng]).addTo(map) .bindPopup('🚤 현재 보트 위치').openPopup(); } else { boatMarker.setLatLng([lat, lng]); } }, function(error) { let message = 'GPS 오류: '; switch(error.code) { case error.PERMISSION_DENIED: message += '위치 접근이 거부되었습니다.
'; break; case error.POSITION_UNAVAILABLE: message += '위치 정보를 사용할 수 없습니다.'; break; case error.TIMEOUT: message += '시간이 초과되었습니다.'; break; default: message += '알 수 없는 오류'; } document.getElementById('info').innerHTML = message; }, { enableHighAccuracy: true, maximumAge: 0, timeout: 15000 } ); } // 페이지 로드되면 바로 GPS 시작 window.onload = startGPS;
